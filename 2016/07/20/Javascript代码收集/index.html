3.模仿jquery each

function each(obj, fn){
    var i;
    if(Object.prototype.toString.call(obj) === '[object Array]'){
        for(i = 0, length = obj.length; i < length; i++){
            fn.call(obj[i], i, obj[i]);
        }
    }
    else if(typeof obj === 'object'){
        for(i in obj){
            if(obj.hasOwnProperty(i)){
               fn.call(obj[i], i, obj[i]);
            }
        }
    }
    else{
        return false;
    }
}
